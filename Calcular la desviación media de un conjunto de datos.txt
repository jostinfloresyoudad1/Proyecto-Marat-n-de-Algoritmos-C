#include <iostream>
#include <vector>
#include <cmath> // Para fabs
#include <iomanip> // Para setprecision y fixed


int main() {
int n;
// 1. Pedir al usuario el tamaño del conjunto y sus elementos
std::cout << "¿Cuántos números ingresarás? ";
std::cin >> n;
if (n <= 0) return 1;
std::vector<double> datos(n);
double suma = 0.0;
std::cout << "Ingresa los " << n << " números:" << std::endl;
for (int i = 0; i < n; ++i) {
std::cin >> datos[i];
suma += datos[i];
}
// 2. Calcular la media
double media = suma / n;
// 3. Inicializar la suma de desviaciones
double sumaDesviaciones = 0.0;
// 4. Recorrer el conjunto de nuevo
for (double x : datos) {
// 5. Sumar la desviación absoluta
sumaDesviaciones += fabs(x - media);
}
// 6. Calcular la desviación media
double desviacionMedia = sumaDesviaciones / n;
// 7. Mostrar el resultado
std::cout << std::fixed << std::setprecision(4);
std::cout << "La desviación media es: " << desviacionMedia << std::endl;
return 0;
}
