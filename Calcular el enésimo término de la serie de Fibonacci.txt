#include <iostream>


int main() {
int n;
long long a = 0, b = 1, siguiente;


std::cout << "Ingresa el término n de la serie Fibonacci a calcular (n >= 0): ";
std::cin >> n;


if (n < 0) {
std::cout << "Error: n debe ser no negativo." << std::endl;
return 1;
}


if (n == 0) {
std::cout << "El término 0 de Fibonacci es: 0" << std::endl;
return 0;
}


for (int i = 2; i <= n; ++i) {
siguiente = a + b;
a = b;
b = siguiente;
}


std::cout << "El término " << n << " de Fibonacci es: " << b << std::endl;


return 0;
}
