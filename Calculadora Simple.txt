#include <iostream>

using namespace std;

int main() {
    double num1, num2;
    char operador;

    cout << "Calculadora Simple" << endl;
    cout << "------------------" << endl;
    cout << "Ingrese el primer numero: ";
    cin >> num1;

    cout << "Ingrese un operador (+, -, *, /): ";
    cin >> operador;

    cout << "Ingrese el segundo numero: ";
    cin >> num2;

    switch (operador) {
        case '+':
            cout << "Resultado: " << num1 << " + " << num2 << " = " << num1 + num2 << endl;
            break;
        case '-':
            cout << "Resultado: " << num1 << " - " << num2 << " = " << num1 - num2 << endl;
            break;
        case '*':
            cout << "Resultado: " << num1 << " * " << num2 << " = " << num1 * num2 << endl;
            break;
        case '/':
            if (num2 != 0) {
                cout << "Resultado: " << num1 << " / " << num2 << " = " << num1 / num2 << endl;
            } else {
                cout << "Error: No se puede dividir por cero." << endl;
            }
            break;
        default:
            cout << "Error: Operador no valido." << endl;
            break;
    }

    return 0;
}
