#include <iostream>
#include <iomanip> // Para setprecision y fixed


int main() {
double suma = 0.0;
int conteo = 0;
int numero;
const int CENTINELA = -1;
std::cout << "Ingresa números para calcular el promedio. Ingresa " << CENTINELA << " para terminar." << std::endl;
// 3. Bucle para pedir números
while (true) {
// 4. Pedir un número
std::cout << "Ingresa un número: ";
std::cin >> numero;
// 5. Verificar si es el centinela
if (numero == CENTINELA) {
break; // Romper el bucle
}
// 6. Sumar y contar
suma += numero;
conteo++;
}
// 7. Verificar si se ingresaron números
if (conteo > 0) {
// 8. Calcular y mostrar el promedio
double promedio = suma / conteo;
std::cout << std::fixed << std::setprecision(2);
std::cout << "El promedio de los " << conteo << " números ingresados es: " << promedio << std::endl;
} else {
std::cout << "No se ingresaron números para calcular el promedio." << std::endl;
}
return 0;
}
