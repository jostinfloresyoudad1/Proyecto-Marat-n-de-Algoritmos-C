#include <iostream>
#include <cstdlib> // Para abs()


int main() {
long long numero;
int conteoDigitos = 0;
// 1. Pedir al usuario que ingrese un número entero
std::cout << "Ingresa un número entero: ";
std::cin >> numero;
// 2. Manejar el caso especial del 0
if (numero == 0) {
std::cout << "El número 0 tiene 1 dígito." << std::endl;
return 0;
}
// 3. Si el número es negativo, convertirlo a positivo
numero = abs(numero);
// 5. Bucle para contar los dígitos
while (numero > 0) {
// 6. Eliminar el último dígito e incrementar el contador
numero /= 10;
conteoDigitos++;
}
// 7. Mostrar el resultado
std::cout << "El número tiene " << conteoDigitos << " dígitos." << std::endl;
return 0;
}
