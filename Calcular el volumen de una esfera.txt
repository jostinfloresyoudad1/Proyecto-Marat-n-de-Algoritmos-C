#include <iostream>
#include <cmath>
#include <iomanip>


int main() {
double radio, volumen;
const double PI = 3.141592653589793;


std::cout << "Ingresa el radio de la esfera: ";
std::cin >> radio;


if (radio <= 0) {
std::cout << "Error: El radio debe ser un valor positivo." << std::endl;
return 1;
}


volumen = (4.0 / 3.0) * PI * pow(radio, 3);


std::cout << std::fixed << std::setprecision(2);
std::cout << "El volumen de la esfera es: " << volumen << std::endl;


return 0;
}
