#include <iostream>


int main() {
int mat1[2][2], mat2[2][2], resultado[2][2];


std::cout << "Ingresa los elementos de la primera matriz 2x2:" << std::endl;
for (int i = 0; i < 2; ++i)
for (int j = 0; j < 2; ++j)
std::cin >> mat1[i][j];


std::cout << "Ingresa los elementos de la segunda matriz 2x2:" << std::endl;
for (int i = 0; i < 2; ++i)
for (int j = 0; j < 2; ++j)
std::cin >> mat2[i][j];


// Sumar las matrices
for (int i = 0; i < 2; ++i)
for (int j = 0; j < 2; ++j)
resultado[i][j] = mat1[i][j] + mat2[i][j];


std::cout << "La matriz suma es:" << std::endl;
for (int i = 0; i < 2; ++i) {
for (int j = 0; j < 2; ++j) {
std::cout << resultado[i][j] << "\t";
}
std::cout << std::endl;
}


return 0;
}
