#include <iostream>
#include <iomanip> // Para std::fixed y std::setprecision


int main() {
int cantidad;
double calificacion, suma = 0.0, promedio;


std::cout << "--- Calculadora de Promedio de Calificaciones ---" << std::endl;
std::cout << "Ingrese la cantidad de calificaciones: ";
std::cin >> cantidad;


if (cantidad <= 0) {
std::cout << "Error: La cantidad de calificaciones debe ser mayor que cero." << std::endl;
return 1; // Termina el programa con un código de error
}


for (int i = 1; i <= cantidad; ++i) {
std::cout << "Ingrese la calificación " << i << ": ";
std::cin >> calificacion;
suma += calificacion;
}


promedio = suma / cantidad;


// Formatear la salida para mostrar 2 decimales
std::cout << std::fixed << std::setprecision(2);
std::cout << "El promedio de las " << cantidad << " calificaciones es: " << promedio << std::endl;


return 0;
}
