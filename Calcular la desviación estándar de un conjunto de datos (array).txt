#include <iostream>
#include <cmath>
#include <vector>


int main() {
int n;
std::cout << "¿Cuántos números ingresarás? ";
std::cin >> n;


if (n <= 0) return 1;
std::vector<double> datos(n);
double suma = 0.0;


std::cout << "Ingresa los " << n << " números:" << std::endl;
for (int i = 0; i < n; ++i) {
std::cin >> datos[i];
suma += datos[i];
}


double media = suma / n;
double sumaDiferenciasCuadrado = 0.0;


for (int i = 0; i < n; ++i) {
sumaDiferenciasCuadrado += pow(datos[i] - media, 2);
}


double varianza = sumaDiferenciasCuadrado / n;
double desviacionEstandar = sqrt(varianza);


std::cout << "La desviación estándar es: " << desviacionEstandar << std::endl;


return 0;
}
