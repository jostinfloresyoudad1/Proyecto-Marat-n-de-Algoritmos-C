#include <iostream>
#include <iomanip> // Para setprecision y fixed


int main() {
double a, r, suma;
int n;
// 1. Pedir al usuario los valores
std::cout << "Ingresa el primer término (a): ";
std::cin >> a;
std::cout << "Ingresa la razón (r): ";
std::cin >> r;
std::cout << "Ingresa el número de términos (n): ";
std::cin >> n;
// 2. Validar que n sea positivo
if (n <= 0) {
std::cout << "Error: El número de términos debe ser positivo." << std::endl;
return 1;
}
// 3. Inicializar suma
suma = 0.0;
// 4. Inicializar término actual
double terminoActual = a;
// 5. Bucle para sumar la serie
for (int i = 0; i < n; ++i) {
// 6. Sumar el término actual
suma += terminoActual;
// 7. Calcular el siguiente término
terminoActual *= r;
}
// 8. Mostrar el resultado
std::cout << std::fixed << std::setprecision(2);
std::cout << "La suma de la serie es: " << suma << std::endl;
return 0;
}
