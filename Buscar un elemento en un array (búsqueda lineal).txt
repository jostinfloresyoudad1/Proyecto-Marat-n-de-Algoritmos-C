#include <iostream>


int main() {
const int TAM = 5;
int numeros[TAM];
int objetivo, posicion = -1;


std::cout << "Ingresa " << TAM << " números enteros:" << std::endl;
for (int i = 0; i < TAM; ++i) {
std::cout << "Número " << i + 1 << ": ";
std::cin >> numeros[i];
}


std::cout << "Ingresa el número que quieres buscar: ";
std::cin >> objetivo;


for (int i = 0; i < TAM; ++i) {
if (numeros[i] == objetivo) {
posicion = i;
break; // Detenemos la búsqueda al encontrar la primera coincidencia
}
}


if (posicion != -1) {
std::cout << "El número " << objetivo << " fue encontrado en la posición (índice) " << posicion << "." << std::endl;
} else {
std::cout << "El número " << objetivo << " no fue encontrado en el array." << std::endl;
}


return 0;
}
